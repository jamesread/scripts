#!/bin/bash
# .profile for James Read <contact@jwread.com>

if type sudo >/dev/null 2>&1 && [ "$UID" -ne 0 ]; then
	sudo="sudo"
else
	sudo=""
fi

# @param $1 source
# @param $2 destination
function getFile() {
	echo "getFile: $1 -> $2"

	if which wget &> /dev/null; then
		wget -nv $1 -O $2
	elif which curl &> /dev/null; then
		curl $1 -o $2 
	else
		echo "Neither wget or curl are available :("
	fi
}

# @param $1 The group.
function fixPerms() {
	$sudo find -exec chown $1:$1 {} +
	$sudo find -exec chmod g+rw {} +
}

if [ -f /etc/bashrc ]; then
	. /etc/bashrc
fi

PATH="/home/$USER/bin/:/usr/bin/:/bin/:/usr/local/bin/:/sbin/:/usr/sbin/:/usr/local/sbin/:/usr/local/sbin/nix/:/usr/share/upsilon-node/tools/"

if [ ! -e "$HOME/.pathexts" ]; then
	touch "$HOME/.pathexts"
fi

while read line; do
	PATH=$PATH:$line
done < "$HOME/.pathexts"

# Setup enviroment variables

# Some hosts have odd values for $TERM
case "$TERM" in
	"dumb"|"")
		TERM="xterm"
esac

# Enviroment variables.
export PATH HOME TERM

# Prompt
case "$TERM" in
	"screen-256color"|"xterm-256color"|"xterm"|"linux"|"screen")
		PS1="\[$(tput setaf 1)\]\u\[$(tput setaf 0)\]\[$(tput sgr0)\]@\[$(tput setaf 6)\]\h\[$(tput sgr0)\]: "
esac

# alias's. 
alias ip='ip -c'
alias ls='ls -F --color'
alias ll='ls -F --color -alh'
alias mkcd='mkcd [ -n "$1" ] && mkdir -p "$@" && cd "$1" '
alias tree='tree -AC'
alias profile-version='tail -n 1 ~/.profile'
alias profile-cptoroot="$sudo cp ~/.profile ~/.bashrc /root"
alias profile-getconf-vim='getFile http://jwread.com/var/nix/vimrc.txt ~/.vimrc '
alias profile-getconf-git='getFile http://jwread.com/var/nix/gitconfig.txt ~/.gitconfig '
alias profile-update='profile_update '
alias profile-puppet-apply-local="$sudo puppet apply --test < /etc/puppet/local.pp"
alias profile-el-install-epel6="$sudo rpm -Uvh http://dl.fedoraproject.org/pub/epel/6/x86_64/epel-release-6-8.noarch.rpm"
alias profile-el-install-epel7="$sudo rpm -Uvh http://dl.fedoraproject.org/pub/epel/7/x86_64/e/epel-release-7-9.noarch.rpm"
alias profile-ownbox='ssh-copy-id $1 && scp ~/.profile $1:~/.profile'
alias profile-ffmpeg-concatscript-mp4='for f in ./*.MP4; do echo "file '\$\f' " >> mylist.txt; done'
alias profile-ffmpeg-go='ffmpeg -f concat -i mylist.txt -c copy output.mkv'
alias profile-rhn-system-id='grep ID- /etc/sysconfig/rhn/systemid'
alias profile-tiff2pdf='for pic in *.tif ; do tiff2pdf $pic > $pic.pdf; done'
alias profile-box-bootstrap='profile_machine_setup'
alias grepria='grep -ria --exclude="\.svn*" '
alias netstatx='netstat -tapue --numeric-hosts --numeric-ports'
alias sctl='systemctl'

function profile_machine_setup() {
	if 
		grep -i 'CentOS Linux release 7' /etc/redhat-release || 
		grep -i 'Red Hat Enterprise Linux release 7' /etc/redhat-release; then
		profile-el-install-epel7
	fi

	$sudo yum install ansible git -y

	profile-git-clone-scripts

	profile-ansible-apply-local
}

function profile-git-clone-scripts() {
	pushd ~
	rm -rf scripts
	git clone https://github.com/jamesread/scripts.git
	popd
}

function profile_update() {
	profile-getconf-git
	profile-getconf-vim

	getFile http://jwread.com/var/nix/profile.txt ~/.profile; 
	source ~/.profile; 
	/bin/cp -f ~/.profile ~/.bashrc; 

	profile-version
}


# support ncurses stuff
export LANG="en_US.utf-8"
export LC_ALL="en_US.utf-8"
export LC_COLORS=1

# make bash suck less
shopt -s checkwinsize

export VISUAL=vim
export SVN_EDITOR=vim

export VAGRANT_DEFAULT_PROVIDER=libvirt

export GOPATH=~/sandbox/go

unset SSH_ASKPASS

# VERSION 1.56
